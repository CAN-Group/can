URL parameters:
start: <float,float>, GPS position, eg. '18.1,53.5'
finish: <float,float>, GPS position, eg. '18.1,53.5'
format: <str>, 'gps', 'kml', or 'geojson'; default: 'gpx'
profile: <str>, see /static/routing_profiles.json for a list of available profiles; default: 'car-fast'
variant: <int>, 0-4; default: 0

example: /api/v2/route?start=18.1,53.5&finish=19.1,54.5&profile=car-fast&variant=1
